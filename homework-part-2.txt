# Функция сортировки
def bubbleSort(arr):
    for a in range(len(arr)):
        for b in range(len(arr) - a):
            if (b+1 < len(arr) and arr[b] > arr[b+1]):
                arr[b], arr[b+1] = arr[b+1], arr[b]

arr = [7, 3, 9, 15, 29, 7, 10]
