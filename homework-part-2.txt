# Меняем местами несколько элементов массива
def swap(arr, a, b):
    temp = arr[a]
    arr[a] = arr[b]
    arr[b] = temp
    

# Функция сортировки
def bubbleSort(arr):
    for a in range(len(arr)):
        for b in range(len(arr) - a):
            if (b+1 < len(arr) and arr[b] > arr[b+1]):
                swap(arr, b, b+1)

arr = [7, 3, 9, 15, 29, 7, 10]
